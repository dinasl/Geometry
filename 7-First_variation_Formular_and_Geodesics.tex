\section{First Variation Formular and Geodesics}
Consider a variation of a $C^{\infty}$ curve $c : [0, l] \to M$ on a Riemannian manifold $M$.

\begin{proposition}
Assumption Let $\epsilon > 0 $ and $C: (-\epsilon, \epsilon) \times [0,l] \to M$
be a $C^{\infty}$ map.
Set 
\[
\begin{aligned}
    c_{s(t)} = c^t(s) := c(s,t), \quad (s,t) -in (-\epsilon, \epsilon) \times [0,l], \\
\end{aligned}
\]
Assume $c_0 (t)$ has constant speed at $s=0$, $\|c_0'(t)\| = \|c_0'(0)\|$, $t\in [0,l]$.
Define the variation vector field $X(t)$ and tangent field $T(s,t)$ by
\[
\begin{aligned}
    X(t) := (c^t)'(0), \quad t \in [0,l] \\
    T(s,t) := c'_{s(t)}, \quad (s,t) \in (-\epsilon, \epsilon) \times [0,l].
\end{aligned}
\]
Note: By definition, $X(t) = dc (\frac{\partial}{\partial s})|_{s=0} \quad T(s,t) = dc(\frac{\partial}{\partial t})$
\end{proposition}

\begin{problem}
    Let $c:[0,l] \to M$ be a $C^{\infty}$ curve and $X(t)$
    a vector field along $c$. Show $\exists$ a variation $\{c_s\}_{s \in (-\epsilon, \epsilon)}$
    such that $c_0 = c $ with variation vector field $X(t)$.
    \label{prob:10}
\end{problem}

\begin{theorem}[First Variation Formula]
    For a variation $\{c_s\}$ of $c_0$ with variation vector field $X$:
    \[
    \begin{aligned}
        \frac{d}{ds} L(c_s)|_{s=0} = \frac{1}{\|c_0'(0)\|} \left( [\langle c_0', C \rangle_{t=0}^{t=l}]- \int_0^l \langle \nabla_{c_0}c_0', X \rangle dt \right)
    \end{aligned}
    \]
\end{theorem}

\begin{proof}
    \[
    \begin{aligned}
        \frac{d}{ds} L(c_s) |_{s=0} = \int_0^l \frac{d}{ds} \frac{d}{ds} \langle c_s'(t),c_s'(t) \rangle^{1/2}|_{s=0} dt \\
        = \frac{1}{2 \|c_0'(0)\|} \int_0^l \frac{d}{ds} \langle c_s'(t), c_s'(t) \rangle |_{s=0} dt \\
    \end{aligned}
    \]
    By \autoref{prop:cov_deriv_linearity_first_arg} $\frac{1}{2} \frac{d}{ds} \langle c_s', c_s' \rangle = \langle \nabla_{c_s'} c_s'(t), c_s'(t) \rangle|_{s=0} = \langle \nabla_{c^t}T, T \rangle |_{s=0}$
    Since $[\frac{\partial}{\partial s}, \frac{\partial}{\partial t}] = 0$, \autoref{prop:cov_deriv_linearity_first_arg} implies
    \[
    \begin{aligned}
        \nabla_{c^t}T = \nabla^c_{\partial / \partial s} dc (\frac{\partial}{\partial s}) = \nabla_{c_0}X
    \end{aligned}
    \]
    Using \autoref{prop:covariant_derivative_leibniz} again,
    \[
    \begin{aligned}
        \langle \nabla_{c_0} X, T \rangle &= \frac{d}{dt} \langle T, X \rangle |_{s=0} - \langle \nabla_{c_0}T, X \rangle \\
        &= \frac{d}{dt} \langle c_0'(t), X \rangle - \langle \nabla_{c_0}, X \rangle
    \end{aligned}
    \]
    Thus
    \[
    \begin{aligned}
        \frac{d}{ds} L(c_s)|_{s=0} &= \frac{1}{\|c_0'(0)\|} \int_0^l \left( \frac{d}{dt} \langle c_0'(t),X \rangle - \langle \nabla_{c_0}c_0', X \rangle\right)dt \\
        &= \frac{1}{\|c_0'(0)\|} \left( [c_0', X]_{t=0}^{t=l} - \int_0^l \langle \nabla_{c_0}c_0', X \rangle dt\right)
    \end{aligned}
    \]
\end{proof}

\begin{proposition}
    If $c$ is a geodisc, it has constant spees, in other word $\|c_{c(t)'}\|$ is independent of $t$.
\end{proposition}

\begin{proof}

\end{proof}

\begin{theorem}
    If a constand speed $C^{\infty}$ curve $c : [0,l] \to M$
    is the shortest among all $C^{\infty}$ curves connecting $p = c(0)$
    and $g = c(l)$, then $c$ is a geodesic.
    \label{thm:7.5}
\end{theorem}

\begin{proof}
    Assume that $c$ is the shortest curve but not a geodesic. Then $\exists t_0 \in (0,l)$ such that $(\nabla_c c')(t_0) \neq 0$.
    Take a local coordinate. nhd $(v, \phi)$ such that $c(t_0) \in U$.
    By continuite, $\exists \delta > 0$ such that $(\nabla_c c')(t) \neq 0$ for $|t-t_0|< \delta$.
    Let $f: [0,l] -to \mathbb{R}$ be a $C^{\infty}$-function such that $f(t)>0$ for $|t-t_0|< \delta$ and $f(t) = 0$ otherwise.
    By \autoref{prob:10}, $\exists$ variation $\{c_s\}$ with variation vector field
    $X(t) := f(t) \cdot \nabla_c c'(t)$. Then :
    $\langle (\nabla_c c')(t), X(t) \rangle = f(t) \| \nabla_c c'(t) \|^2$
    which is positive for $|t-t_0|< \delta$ and zero otherwise.
    By the First Variation Formula:
    \[
    \begin{aligned}
        \frac{d}{dt} L(c_s)|_{t=0} = - \frac{1}{\|c'(0)\|} \int_{t_0-\delta}^{t_0 + \delta} f(t) \|(\nabla_c c')(t)\|^2 dt <0
    \end{aligned}
    \]
    This contradicts the minimality of $L(c_0)$
\end{proof}
For a coordinate chart $(U, \phi)$ of $M$, Let
$\phi(c(t)) = (c^1(t), \dots, c^n(t))$. Then $\nabla_c c' = 0$ is:
\[
\begin{aligned}
    \frac{d^2c^k}{dt^2} + \sum_{ij=1}^n \Gamma_{ij}^k \frac{dc^i}{dt} \cdot \frac{dc^j}{dt} = 0, \quad k = 1, \dots, n
\end{aligned}
\]
Rewriting with $X^k = \frac{dc^k}{dt}$, we have
\[
\begin{aligned}
    \frac{d X^k}{dt} + \sum_{ij = 1}^n \Gamma_{ij}^k X^i (t) X^j (t) = 0
\end{aligned}
\]
This is a first order nonliner ordiner differential equation.

\begin{proposition}
    $\forall p \in M$ and $\forall v \in T_p M$, $\exists \delta>0$ and a unique geodesic
    $c: (-\delta, \delta) \to M$ such that $c(0) = p$ and $c'(0) = v$.
    \label{prop:7.6}
\end{proposition}

\begin{proposition}
    Let $a<0<b$. If two geodesics $\sigma, \tau : (a,b) \to M$ satisfy $\sigma(0) = \tau(0)$ and $\sigma'(0) = \tau'(0)$, then $\sigma(t) = \tau(t) \quad \forall t \in (a,b)$.
    \label{prop:7.7}
\end{proposition}

\begin{proof}
    Show $\sigma(t) = \tau(t)$ for $t \in [0,b)$.
    Let
    \[
    \begin{aligned}
        t_0 : \sup \{s \in [0,b) | \sigma(t) = \tau(t) \forall t \in [0,s)\}
    \end{aligned}
    \]
    By \autoref{prop:7.6}, $t_0>0$. Assume $t_0 <b$.
    Let $t_i := t_0 - \frac{1}{i} \quad(i >> 0)$
    Since $\sigma,\tau$ are continous and $\sigma(t_i) = \tau(t_i)$, then $\sigma(t_0) = \tau(t_0)$.
    $\sigma = \tau$ on $[0, t_0] \implies \sigma'(t_0)=\tau'(t_0)$ 
    By \autoref{prop:7.6}, $\sigma= \tau$ in a nhd of $t_0$, contradicting the definition of $t_0$
\end{proof}

\begin{definition}
    A geodesic $c(t)$ extended to its maximal possible somain in both directions is a maximal geodesic.
    \label{def:7.8}
\end{definition}

For $p \in M$, $v \in T_p M$ ,let $\gamma_v : J_v \to M$ be the maximal geodesic such thath 
$\gamma_v(0) = p$, $\gamma_v'(0) = v$. Note:

\begin{enumerate}
    \item $J_v \subset \mathbb{R}$ is an open interval containing $0$.
    \item $\forall a,t \in \mathbb{R}$, $\forall v \in TM$ \\
    $t\in J_{av} \equiv at \in J_v$ and $\gamma_{av}(t) = \gamma_v(at)$
    \item $\|\gamma_v'(t)\| = const$ $\implies$ solution of the geodesic equation do not explode within coordinate charts and extend to the boundary.
\end{enumerate}

\begin{proposition}
    For $\{t_i\}_{i=1}^{\infty} \subset J_v$, if $t_i \to \sup{J_v}$ (or $\inf{J_v}$), 
    then  $\{\gamma_v(t_i)\}_{i=1}^{\infty}$ does not converge in $M$.
    \label{prop:7.9}
\end{proposition}

\begin{proposition}
    $\forall p \in M,\quad v \in T_pM, \quad t_0 \in J_V$,
    $\exists O \subset TM : \text{neighbourhood of } v$ such that $\forall w \in O, t_0 \in J_w$.
    Moreover, 
    \[
    \begin{aligned}
        TM &\to M \text{ is } C^{\infty} \\
        v &\to \gamma_v(t_0)
    \end{aligned}
    \]
    \label{prop:7.10}
\end{proposition}

\begin{problem}
    Let $(\bar{M}, \bar{g})$ be a Riemannian manifold, $M \subset \bar{M}$ a submanifold,
    and $\bar{\nabla}, \nabla$ be the respective Levi-Cevita connections.
    $\forall p \in M$, let $\pi: T_p \bar{M} \to T_pM$ be the orthogonal projection.
    \textbf{Show}: $(\nabla_X Y)_p = \pi ((\bar{\nabla}_X Y)), \quad X,Y \in \mathbb{X}(M)$
    \label{prob:11}
\end{problem}

\begin{example}
    Find the geodesic of $\gamma$ with arc-lenght parameter in the 2-dimensional unit sphere 
    $\mathbb{S}^2 \subset \mathbb{R}^3$. By \autoref{prob:11}.
    \[
    \begin{aligned}
        \nabla_{\gamma} \gamma' = \pi (\gamma''(t)) = 0,
    \end{aligned}
    \]
    here, $\pi : \mathbb{R}^3 \to T_p \mathbb{S}^2, \quad (p \in \mathbb{S}^2)$ is the orthogonal projection.
    This implies that $\gamma''(t)$ is parallel to the line segment connecting $O$ with $\gamma(t)$. Thus $\exists f(t)$ such that
    $\gamma''(t) = f(t) \cdot \gamma(t)$.
    Differentiating $\|\gamma(t)\|^2_{\mathbb{R}^3} = 1$ twice,
    \[
    \begin{aligned}
        0 =& \frac{d}{dt} \langle \gamma(t), \gamma(t) \rangle_{\mathbb{R}^3} = 2 \langle \gamma'(t), \gamma(t) \rangle_{\mathbb{R}^3} \\
        &f(t) = \langle \gamma''(t), \gamma(t) \rangle_{\mathbb{R}^3} = - \langle \gamma'(t), \gamma'(t) \rangle_{\mathbb{R}^3} = -1
    \end{aligned}
    \]
    We obtain $\gamma''(t) + \gamma(t) = 0$. 

    $\gamma(t) = cos(t) \gamma(0) + sin(t) \gamma'(0)$ which represents a great circle.
\end{example}

\end{problem}
