\section{Riemannian metric}

Assume that $M$ is Hansforff differentiable manifold.
\begin{definition}[Riemannian metric and Riemannian manifold]
    A \emph{Riemannian metric} $g$ is a map
    \[
        M \owns p \to \langle \cdot, \cdot \rangle_p =: g_p ( \cdot, \cdot ) \text{that assigns a},
    \]
    (postive definite) inner product to each $T_pM$.
    The pair $\left(M, g\right)$ is called a \emph{Riemannian manifold}.
    In terms of tensor field, $g$ is a $(0,2)$-type tensor field (or a 2-form) on $M$.
\end{definition}

\begin{definition}[Length and angle]
For $v,w \in T_p M$, we have the following notions of length and angle:
\begin{itemize}
    \item \emph{Length} $\|v\| := \sqrt{\langle v, v \rangle}$.
    \item \emph{Angle} : $\theta \in [0,\pi]$ such that $cos \theta = \frac{\langle v, w \rangle}{\|v\| \|w\|}$.
\end{itemize}
\end{definition}

\begin{example}
Let $(v, \phi)$ be a chart with
\[
\phi = (x^1, \dots, x^n).
\]
Define $g_{ij} : U \to \mathbb{R}$ by
\[
g_{ij}(p) :=  \left\langle \left(\frac{\partial}{\partial x^i}\right)_p, \left(\frac{\partial}{\partial x^j}\right)_p \right\rangle.
\]
On $U$, $g$ is expressed as 
\[
g = \sum_{i,j=1} g_{ij} dx^i \otimes dx^j.
\]
The matrix $(g_{ij})$ is a postive definite symmetric matrix.
\[
ds^2 = \sum_{i,j =1}^n \delta_{ij} dx^i \otimes dx^j = \sum_{i=1}^n (dx^i)^2.
\]
\end{example}

\begin{problem}
Consider two charts $(U, \phi)$ and $(V, \psi)$ with $U \cap V \neq \emptyset$. 
\[
\phi = (x^1, \dots, x^n), \psi = (y^1, \dots, y^n),
\]
and let $g_{ij}$, $\hat{g}_{kl}$ be the components of g with respect to these charts.
Show that on $U \cap V$ satisfies the following transformation law:
\[
\hat{g}_{kl} = \sum_{i,j=1}^n g_{ij} \frac{\partial x^i}{\partial y^k} \frac{\partial x^j}{\partial y^l}.
\]
\end{problem}

\begin{definition}[$C^{\infty}$ Riemannian manifold]
$(M,g)$ is a \emph{$C^{\infty}$ Riemannian manifold} if $g$ is a $C^{\infty}$ tensor field.
\end{definition}
\begin{proposition}
The following are equivalent:
\begin{itemize}
    \item (M,g) is a $C^{\infty}$ Riemannian manifold.
    \item Copmonents $g_{ij}$ are $C^{\infty}$ functions for any chart $(U, \phi)$.
    \item $p \to \langle X_p, Y_p \rangle$ is $C^{\infty}$ for any $C^{\infty}$ vector fields $X,Y$.
\end{itemize}    
\end{proposition}
Henceforth, assume that all Riemanninan metrics are $C^{\infty}$.
\begin{remark}
$g,h$ : Riemannian metric, $f \in c^{\infty} ( M, \mathbb{R}_{>0})$
$\implies$ $fg$ and $g+h$ are also Riemannian metrics.
\end{remark}

\begin{proposition}
    Every paracompact manifold admits a Riemannian metric.
\end{proposition}

\begin{proof}
    Let $\{(U_k, \phi_k)\}_{k=1}^{K_0}$ be a finite atlas and let $\{x_k\}^{K_0}_{k=1}$
    be a partition of unity subordinate to ${U_k}$. On $U_k$, define $g_k$
    \[
    g_k(u,w) := \langle(d \phi_k)_p(v), (d \phi_k)_p (w) \rangle_{\mathbb{R}^n}, v,w \in T_pM.
    \]
    Then the global metric is given by:
    \[
    g := \sum_{k=1}^{K_0} x_k g_k
    \]
\end{proof}

\subsection{Isometries}

\begin{definition}[Isometry]
Let $(M,g)$ and $(N,h)$ be Riemannian manifolds and let $f : M \to N$ be a smooth map.
\begin{itemize}
\item The map $f$ is called \emph{locally isometric} if
\[
\langle (df)_p(v), (df)_p(w) \rangle_{f(p)} = \langle v, w \rangle_p
\quad \text{for all } p \in M, \; v,w \in T_pM.
\]
\item The map $f$ is called an \emph{isometry} if it is a diffeomorphism and locally isometric.
\item The manifolds $M$ and $N$ are called \emph{isometric} if there exists an isometry $f : M \to N$.
\end{itemize}
\end{definition}

\begin{proposition}
\begin{enumerate}[label=(\arabic*)]
\item $f: L \to M, h : M \to N$ are (local) isometric maps $\implies h \circ f$ is a (local) isometric map.
\item $f$ is isometry $\implies f^{-1}$ is isometry.
\end{enumerate}
Thus, the relation "Isometric" is an equivalence relation.
\end{proposition}

\subsection{Riemannian submanifold}

\begin{definition}[Riemannian submanifold]
Assume $M$ is a Riemannian manifold and $N \subset M$ is a submanifold.
$N$ is a \emph{Riemannian submanifold} if the metric is defined by $\langle v, w \rangle_N := \langle v, w \rangle_M$ for all $v, w \in T_pN$.
\end{definition}

The inclusion $C : N \to M$ is always an isometric map.

\subsubsection{Remember 3.10 Nash Embedding Theorem}
Every Riemannian manifold is isometric to a riemannian submanifold of some Euclidian space $\mathbb{R}^n$. $\Leftrightarrow$
\emph{Whitney Embedding Theorem}

\subsection{Riemannian product and warped product}

\begin{definition}[Riemannian product and product Riemannian manifold]
Let $M, N$ be Riemannian manifolds. The \emph{Riemannian product} $M \times N$ is defined by
\[
T_{(p,q)} (M \times N) \simeq T_p M \times T_q N, \quad (p,q) \in M \times N.
\]
The metric is defined by
\[
\langle v, w \rangle_{M \times N} := \langle v_1, w_1\rangle + \langle v_2, w_2 \rangle, \quad v = (v_1, v_2), w = (w_1, w_2) \in T_pM \times T_qN.
\]
We call $(M\times N, \langle \cdot, \cdot \rangle_{M \times N})$ the \emph{product Riemannian manifold} (of $M$ and $N$).

In this metric, $T_pM$ and $T_qN$ are orthonormal to each other and each factor remains its original inner product.
\end{definition}

\begin{example}
The Riemannian product $\mathbb{R}^m \times \mathbb{R}^n$ is isometric to $\mathbb{R}^{m+n}$.
\end{example}

\begin{definition}[Warped product]
Let $M, N$ be Riemannian manifolds and let $f : M \to \mathbb{R}_{>0}$ be a $C^{\infty}$ function. 
The \emph{warped product} $M \times_f N$ is teh product manifold $M\times N$ equipped with the metric
\[
\langle v, w \rangle_{M \times_f N} := \langle v_1, w_1\rangle + f(p)^2 \langle v_2, w_2 \rangle, \quad v = (v_1, v_2), w = (w_1, w_2) \in T_{(p,q)}(M \times N).
\]
We call the function $f$ the \emph{warping function}.
\end{definition}

\begin{problem}
Let $(0, +\infty)$ have the Euclidean metric and $\mathbb{S}^{n-1} \in \mathbb{R}^n$ have the standard metric.
Show that the warped product $(0, +\infty) \times_r \mathbb{S}^{n-1}$ is isometric to $(\mathbb{R}^n \setminus \{0\},g_{\mathbb{R}^n})$,
where $r(x) := x,\quad x \in (0, +\infty)$ is the warping function. 
\end{problem}

\begin{definition}[$C^{\infty}$ transformation group]
A \emph{$C^{\infty}$ transformation group} $P = \{\gamma_i\}_{i\in I}$ on a Riemannian manifold $M$ is a group of isometries if each $\gamma_i \in P$ is an isometry.
\end{definition}

\langle v, w \rangle_{M/\Gamma} := \langle d(\pi|_{U_p}^{-1})(v), d(\pi|_{U_p}^{-1})(w) \rangle, \quad v,w \in T_{[p]}(M/\Gamma).

\subsection{Volume element}

Let $M$ be an oriented $n$-diemnsional Riemannian manifold.
\begin{definition}
For a positively oriented chart $(U, \phi)$ with $\phi = (x^1, \dots, x^n)$, the n-form on
\[
\operatorname{dvol} := \sqrt{|g|}dx^1 \wedge \dots \wedge dx^n, \quad |g| := \operatorname{det}(g_{ij}),
\]
is called the \emph{volume element} (or \emph{volume form}) of $M$. 
\end{definition}

\begin{remark}
$\operatorname{dvol}$ is a symbol and not necessarly the exterior derivative or some $(n-1)$-form.
This form is globally well-defined on $M$.
In fact, if $(U,\phi), (V,\psi)$ are two positively oriented charts on $M$, then on $U\cap V(\neq \emptyset)$,
\[
\begin{aligned}
&\operatorname{det}(\hat{g}_kl) = \operatorname{det}\left(\frac{\partial x^i}{\partial y^k}\right)^2 \operatorname{det}(g_{ij}) \\
\implies &\sqrt{|\hat{g}|} = \left|\operatorname{det}\left(\frac{\partial x}{\partial y}\right)\right| \sqrt{|g|} \text{ and} \\
& dy^1 \wedge \dots \wedge dy^n = \operatorname{det}\left(\frac{\partial y}{\partial x}\right) dx^1 \wedge \dots \wedge dx^n.
\end{aligned}
\]
Thus $\sqrt{|\hat{g}|} dy^1 \wedge \dots \wedge dy^n = \sqrt{|g|} dx^1 \wedge \dots \wedge dx^n$.
\end{remark}

\begin{definition}[Volume]
The \emph{volume} of an open subset $D \subset M$ is defined by:
\[
\operatorname{vol}(D) := \int_D \operatorname{dvol}.
\]
\end{definition}

\begin{remark}
For non-orientable manifolds, the volume can still be defined by partitioning $D$ into small orientable regions.
\end{remark}
