\section{Riemannian metric}

Assume that $M$ is Hansforff differentiable manifold.

\begin{definition}
    A Riemannian metric $g$ is a map
    $$
        M \owns p \to \langle \cdot, \cdot \rangle_p =: g_p ( \cdot, \cdot ) \text{that assigns a}
    $$

    (postive definite) inner product to each $T_pM$.

    The pair $\left(M, g\right)$ is called a Riemannian manifold.

    In terms of tensor field, g is a $(0,2)$- type tensor field (or a 2-form) on M
\end{definition}

\begin{definition}
    For $v,w \in T_p M$
    \begin{itemize}
        \item Length $\|v\| := \sqrt{\langle v, v \rangle}$
        \item Angle : $\theta \in [0,\pi]$ such that $cos \theta = \frac{\langle v, w \rangle}{\|v\| \|w\|}$
    \end{itemize}
\end{definition}

Let $(v, \phi)$ be a chart with
\[
\begin{aligned}
\phi = &(x^1, \dots, x^n). \text{ Define } g_{ij} : U \to \mathbb{R} \text{ by } \\
 &g_{ij}(p) :=  \langle \left(\frac{\partial}{\partial x^i}\right)_p, \left(\frac{\partial}{\partial x^j}\right)_p \rangle.
\end{aligned}
\]


On $U$, $g$ is expressed as 
\[
g = \sum_{i,j=1} g_{ij} dx^i \otimes dx^j.
\]

The matrix $(g_{ij})$ is a postive definite symmetric matrix.

\[
ds^2 = \sum_{i,j =1}^n \delta_{ij} dx^i \otimes dx^j = \sum_{i=1}^n (dx^i)^2
\]

\begin{problem}
    Consider two charts $(U, \phi)$ and $(V, \psi)$ with $U \cap V \neq \emptyset$. 

    \[
    \phi = (x^1, \dots, x^n), \psi = (y^1, \dots, y^n)
    \]

    let $g_{ij}$, $\hat{g}_{kl}$ be the components of g with respect to these charts.

    Show that on $U \cap V$,

\[
\hat{g}_{kl} = \sum_{i,j=1}^n g_{ij} \frac{\partial x^i}{\partial y^k} \frac{\partial x^j}{\partial y^l}.
\]
\end{problem}

\begin{definition}
    (M,G) is a $C^{\infty}$ Riemannian manifold

    \[
    \logeq g \text{ is a } c^{\infty} \text{ tensor field}
    \]

    The following are equivalent:
    \begin{itemize}
        \item (M,g) is a $C^{\infty}$ Riemannian manifold
        \item Copmonents $g_{ij}$ are $C^{\infty}$ functions for any chart $(U, \phi)$
        \item $p \to \langle _p, Y_P \rangle$ is $C^{\infty}$ for any $C^{\infty}$ vector fields $X,Y$.
    \end{itemize}    
\end{definition}
    Henceforth, assume that all Riemanninan metrics are $C^{\infty}$

    $g,h$ : Riemannian metric, $f \in c^{\infty} ( M, \mathbb{R}_{>0})$

    $\implies$ $fg$ and $g+h$ are also Riemannian metrics.

\begin{proposition}
    Every paracompact manifold admits a Riemannian metric.
\end{proposition}

\begin{proof}[case : M is cpt]
    Let $\{(U_k, \phi_k)\}_{k=1}^{K_0}$ be a finite atlas. let $\{x_k\}^{K_0}_{k=1}$

    be a partition of unity subordinate to ${U_k}$/

    On $U_k$, define $g_k$

    \[
    g_l(u,w) := \langle(d \phi_k)_p(v), (d \phi_k)_p (w) \rangle_{\mathbb{R}^n}, v,w \in T_pM
    \]

    Then the global metric is given by:

    \[
    g := \sum_{k=1}^{K_0} x_k g_k
    \]
\end{proof}


\subsection{Isometries}

\begin{definition}
    Let $M, N$ be Riemannian manifoolds and $f: M \to N$ be a $C^{\infty}$ map

    \begin{itemize}
        \item f is locally isometric map $\logeq \langle (df)_p(v), (df)_p(w) \rangle = \langle v, w \rangle \forall v,w \in T_pM, p \in M$
        \item An injective locally isometric map is an isometric map.
        \item An isometric map that is also a diffeomorphism is an isometry.
        \item $M$ and $N$ are isometric if there exists an isometry between them.
    \end{itemize}
\end{definition}

\begin{proposition}[(1)]
    $f: L \to M, h : M \to N$ are (local) isometric maps $\implies h \circ f$ is a (local) isometric map.

    [(2)] $f$ is isometry $\implies f^{-1}$ is isometry.
    
    hus, the relation "Isometric" is an equivalence relation.
\end{proposition}

\subsection{Riemannian submanifold}

\begin{definition}
    $M:$ Riemannian manifold, $N \subset M$ is a submanifold.

    $N$ is a Riemannian submanifold if the metric is defined by $\langle v, w \rangle_N := \langle v, w \rangle_M$ for all $v, w \in T_pN$.
\end{definition}

The inclusion $C : N \to M$ is always an isometric map.

\subsubsection{Remember 3.10 Nash Embedding Theorem}
Every Riemannian manifold is isometric to a riemannian submanifold of some Euclidian space $\mathbb{R}^n$. $\Leftrightarrow$
\emph{Whitney Embedding Theorem}
